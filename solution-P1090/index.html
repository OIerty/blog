【洛谷】P1090【合并果子】

这乍看，很水的贪心：每次将最小两队合并为一堆

再一编：TLE我去

于是就得用到优先队列

本人不是很懂，请教了旁人

C++代码如下：

    #include<bits/stdc++.h>
    using namespace std;
    long long n,s,t;
    priority_queue<int, vector<int>,greater<int> >q;//优先队列 
    int main(){
    	cin>>n;
    	for(int i=0;i<n;i++){
    		cin>>t;
    		q.push(t);//入
    	}
    	for(int i=n;i>1;i--){
    		t=q.top();
    		q.pop();//出
    		t+=q.top();//单次力
    		q.pop();
    		s+=t;//加入总值
    		q.push(t);//入合并好的堆
    	}
    	cout<<s;//输出
    	return 0;
    }
